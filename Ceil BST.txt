typedef long long int ll;
struct node
{
    struct node *lchild,*rchild;
    ll data;
}*root=NULL;

void insert(ll x)
{
    struct node *p=new node;
    if(root==NULL)
    {
        p->lchild=p->rchild=NULL;
        p->data=x;
        root=p;
    }
    else
    {
        struct node *q,*r;
        q=root;
        r=NULL;
        ll k=0;
        while(q!=NULL)
        {
            r=q;

             q=q->lchild;
            else if(q->data<x)
             q=q->rchild;
            else
            {
                k=1;
                break;
            }
        }

            p->data=x;
            if(r->data>x)
             r->lchild=p;
            else
             r->rchild=p;
        }
    }
}
